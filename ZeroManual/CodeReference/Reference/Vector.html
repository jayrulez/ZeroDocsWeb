class VectorMathDemo : ZilchComponent
{
    function Initialize(init : CogInitializer)
    {
        // Note: Vector functions that have the following calling convention,
        //        Math.Function(vec): vec, do not modify "vec", but returns the solution
        // Note: There is no "Vector" class, they are simply represented by the built-in
        //       Real3 class, where the Vector functionality is contained within the 
        //       Math library. 

        // Vector construction.
        var vector0 = Real3();

        // Vector assignment.
        var vector1 = Real3(1, 1, 1);

        // Returns a normalized version of the given vector.
        var vectorNrm = Math.Normalize(vector1);

        // Returns a float equal to the length of a vector.
        var vectorLength = Math.Length(vector1);

        // Returns the result of the cross product between "vector0" and "vector1".
        // Reminder: The result is a vector perpendicular to both "vector0" and "vector1".
        var vectorCrossProduct = Math.Cross(vector0, vector1);

        // Returns the result of the dot product between "vector0" and "vector1".
        // Reminder: The result is a float that can be used in relating the angle
        //           between the two vectors and projection.
        var vectorDotProduct = Math.Dot(vector0, vector1);

        // Reflects "vector0" about "vector1". See awesome ASCII art:
        //
        //             vector1
        //  vector0       ^       vectorReflect
        //        \       |       /
        //         \      |      /
        //          \     |     /
        //           \    |    /
        //            \   |   /
        //             \  |  /
        //              \ | /
        // ______________\|/________________
        //
        var vectorReflect = Math.Reflect(vector0, vector1);

        // Projects "vector0" onto "vector1". See awesome ASCII art:
        //
        //           vector0
        //          /
        //         /
        //        /                 vector0 ____
        //       /                  becomes     |
        //      /                               |
        //     /                                V
        //    /                           vectorProject
        //   /                               ___|___
        //  /                               /       \     vector1
        // /--------------> vector1         --------->---->
        //
        var vectorProject = Math.Project(vector0, vector1);

        // Linearly interpolate between the two vectors using the function:
        // vectorLerp = vector0 + t * (vector1 - vector0)
        var t = 0.5;
        var vectorLerp = Math.Lerp(vector0, vector1, t);

        // Gets the angle of the vector about the z-plane. Think "unit circle angle".
        var vectorAngleZ = Math.Angle2D(vector1);

        // Returns the angle between the two vectors in radians.
        var vectorAngle = Math.AngleBetween(vector0, vector1);

        // Gets the angle between two vectors, preserves the sign of the angle.
        // Needs a 3rd vector to build a basis in order to make the "sign" have
        // any sort of meaning.
        var upVector = Real3(0, 1, 0);
        var vectorSignedAngle = Math.SignedAngle(vector0, vector1, upVector);

        // Rotates "vector0" toward "vector1" by, at most, "maxAngle" (in radians).
        var vectorRotateTowards = Math.RotateTowards(vector0, vector1, 1);

        // Rotates "vector0" about "zAxis" by half pi (90 degrees).
        var zAxis = Real3(0, 0, 1);
        var rotatedVector = Math.RotateVector(vector0, zAxis, 1.5707);
    }
}

import Zero;
import Events;
import Property;
import VectorMath;
Vec3 = VectorMath.Vec3;

class VectorBasics:
    
    def Initialize(self, initializer):
        # Note: Vector functions that have the following calling convention,
        #       "vector.function()", do not modify "vector".
        # Note: Python does not support true value types. This is normally not
        #       a problem because most value types (strings, numbers) are not
        #       mutable. So when a vector is passed to a function you can modify
        #       the original value by assigning to its individual elements or 
        #       calling normalize.

        # Vector assignment.
        vector0 = Vec3(0, 0, 0);

        # Also vector assignment.
        vector1 = VectorMath.Vec3(1, 1, 1);

        # Normalizes "vector0" in place, that is once the function is called on
        # the vector, the vector will have a length of 1.
        vector0.normalize();

        # Returns a normalized version of the given vector.
        vectorNrm = vector1.normalized();

        # Returns a float equal to the length of a vector.
        vectorLength = vector1.length();

        # Copies a vector.
        vectorCopy = vector0.copy();

        # Returns the result of the cross product between "vector0" and "vector1".
        # Reminder: The result is a vector perpendicular to both "vector0" and "vector1".
        vectorCrossProduct = vector0.cross(vector1);

        # Alternate syntax for the cross product operation.
        vectorCrossProduct = VectorMath.Vec3.cross(vector0, vector1);

        # Returns the result of the dot product between "vector0" and "vector1".
        # Reminder: The result is a float that can be used in relating the angle
        #           between the two vectors and projection.
        vectorDotProduct = vector0.dot(vector1);

        # Alternate syntax for the dot product operation.
        vectorDotProduct = VectorMath.Vec3.dot(vector0, vector1);

        # Reflects "vector0" about "vector1". See awesome ASCII art:
        #
        #             vector1
        #  vector0       ^       vectorReflect
        #        \       |       /
        #         \      |      /
        #          \     |     /
        #           \    |    /
        #            \   |   /
        #             \  |  /
        #              \ | /
        # ______________\|/________________
        #
        vectorReflect = vector0.reflect(vector1);

        # Alternate syntax for the reflect operation.
        vectorReflect = VectorMath.Vec3.reflect(vector0, vector1);

        # Projects "vector0" onto "vector1". See awesome ASCII art:
        #
        #           vector0
        #          /
        #         /
        #        /                 vector0 ____
        #       /                  becomes     |
        #      /                               |
        #     /                                V
        #    /                           vectorProject
        #   /                               ___|___
        #  /                               /       \     vector1
        # /--------------> vector1         --------->---->
        #
        vectorProject = vector0.project(vector1);

        # Alternate syntax for the project operation.
        vectorProject = VectorMath.Vec3.project(vector0, vector1);

        # Linearly interpolate between the two vectors using the function:
        # vectorLerp = vector0 + t * (vector1 - vector0)
        t = 0.5
        vectorLerp = vector0.lerp(vector1, t);

        # Gets the angle of the vector about the z-plane. Think "unit circle angle".
        vectorAngleZ = vector1.angleZ();

        # Returns the angle between the two vectors in radians.
        vectorAngle = VectorMath.Vec3.Angle(vector0, vector1);

        # Gets the angle between two vectors, preserves the sign of the angle.
        # Needs a 3rd vector to build a basis in order to make the "sign" have
        # any sort of meaning.
        upVector = Vec3(0, 1, 0);
        vectorSignedAngle = VectorMath.Vec3.SignedAngle(vector0, vector1, upVector);

        # Rotates "vector0" toward "vector1" by, at most, "maxAngle" (in radians).
        vectorRotateTowards = VectorMath.Vec3.RotateTowards(vector0, vector1, 1);

        # Rotates "vector0" about "zAxis" by half pi (90 degrees).
        zAxis = Vec3(0, 0, 1);
        rotatedVector = VectorMath.Vec3.RotateVector(vector0, zAxis, 1.5707);

        return;

Zero.RegisterComponent("VectorBasics", VectorBasics)
