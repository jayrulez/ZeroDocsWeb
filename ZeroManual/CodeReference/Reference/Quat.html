class QuaternionDemo : ZilchComponent
{
    function Initialize(init : CogInitializer)
    {
        // Note: Quaternion functions that have the following calling convention,
        //       Math.Function(quat): quat, do not modify "quat", but returns the solution
        //       Also, for quaternions to do any kind of useful rotations, 
        //       they must be unit length.

        // Quaternion construction.
        var quat0 = Quaternion();
        
        // Quaternion assignment.
        var quat1: Quaternion = Quaternion.Identity;


        // Returns a normalized version of the given quaternion.
        var quatNrm = Math.Normalize(quat0);

        // Returns an inverted copy of the quaternion, one that does the opposite
        // rotation of the original quaternion.
        var quatInv = Math.Invert(quat0);
        
        // Performs a spherical linear interpolation between two quaternions and
        // returns the interpolated rotation at the given t-value.
        var quatSlerp = Math.Slerp(quat0, quat1, 0.5);

        // Rotates a 3D vector by the quaternion.
        var vector0 = Real3(1, 1, 1);
        var vecRotated = Math.Transform(vector0, quat0);

        // Rotates the quaternion towards the given quaternion by, at most, the
        // max radians passed in.
        // (quatRotate is quat0 rotated toward quat1 by, at most, 90 degrees)
        var quatRotate = Math.RotateTowards(quat0, quat1, 1.5708);

        // Create a quaternion from an axis-angle pair.
        var vecAxis = Real3(1, 0, 0);
        var angle = 1.5707;
        var quatAxisAngle = Math.ToQuaternion(vecAxis, angle);
        
        // Alternate version of the above.
        quatAxisAngle = Math.AxisAngle(vecAxis, angle); 

        // Creates a quaternion representing the rotation where the "facing" is
        // the forward vector and the "up" is the up vector. Makes the most sense
        // when applied as a camera's orientation.
        var vecFacing = Real3(0, 0, 1);
        var vecUp = Real3(0, 1, 0);
        var quatFacing = Math.ToQuaternion(vecFacing, vecUp);

        // Creates the quaterion that rotates from "vecAxisX" to "vecAxisZ".
        var vecAxisX = Real3(1, 0, 0);
        var vecAxisZ = Real3(0, 0, 1);
        var quatAtoB = Math.RotationQuaternionBetween(vecAxisX, vecAxisZ);

        // Creates a quaternion from the given Euler angles (in radians).
        // The angles here are 90, 60, and 30 degrees, respectively.
        var quatEuler = Math.ToQuaternion(1.5707, 1.0472, 0.5236);

        // Creates a quaternion from a vector representing Euler angles (in radians).
        var eulerAngles = Real3(1.5707, 1.0472, 0.5236);
        quatEuler = Math.Euler(eulerAngles);

        // Alternate version of the above:
        quatEuler = Math.ToQuaternion(eulerAngles);

        // Gets the angle between two rotations. Quaternions must be unit length
        // for the angle to make any sense.
        var angleDiff = Math.AngleBetween(quat0, quat1);

        // To combine two quaternions (combining two rotations into one), use 
        // the multiplication operator. You'll need to normalize this result to
        // guarantee a valid rotation. Order matters! quat0 * quat1 != quat1 * quat0
        var combinedRot = Math.Transform(quat0, quat1);        
    }

}

import Zero;
import Events;
import Property;
import VectorMath;
Vec3 = VectorMath.Vec3;
Quat = VectorMath.Quat;

class QuatBasics:
    
    def Initialize(self, initializer):
        # Note: Quaternion functions that have the following calling convention,
        #       "quat.function()", do not modify "quat".
        #       Also, for quaternions to do any kind of useful rotations, they
        #       must be unit length.

        # Quaternion assignment.
        quat0 = Quat();

        # Also quaternion assignment.
        quat1 = VectorMath.Quat();

        # Returns a normalized version of the given quaternion.
        quatNrm = quat0.normalized();

        # Returns an inverted copy of the quaternion, one that does the opposite
        # rotation of the original quaternion.
        quatInv = quat0.inverted();

        # Performs a spherical linear interpolation between two quaternions and
        # returns the interpolated rotation at the given t-value.
        quatSlerp = quat0.slerp(quat1, 0.5);

        # Alternate calling convention of the above function.
        quatSlerp = VectorMath.Quat.Slerp(quat0, quat1, 0.5);

        # Copies a quaternion.
        quatCopy = quat1.copy();

        # Rotates a 3D vector by the quaternion.
        vector0 = Vec3(1, 1, 1);
        vecRotated = quat0.rotate(vector0);

        # Alternate calling convention of the above function.
        vecRotated = VectorMath.Quat.RotateVector(quat0, vector0);

        # Rotates the quaternion towards the given quaternion by, at most, the
        # max radians passed in.
        # (quatRotate is quat0 rotated toward quat1 by, at most, 90 degrees)
        quatRotate = quat0.towards(quat1, 1.5708);

        # Alternate calling convention of the above function.
        quatRotate = VectorMath.Quat.RotateTowards(quat0, quat1, 1.5708);

        # Create a quaternion from an axis-angle pair.
        vecAxis = Vec3(1, 0, 0);
        angle = 1.5707;
        quatAxisAngle = VectorMath.Quat.AxisAngle(vecAxis, angle);

        # Creates a quaternion representing the rotation where the "facing" is
        # the forward vector and the "up" is the up vector. Makes the most sense
        # when applied as a camera's orientation.
        vecFacing = Vec3(0, 0, 1);
        vecUp = Vec3(0, 1, 0);
        quatFacing = VectorMath.Quat.Facing(vecFacing, vecUp);

        # Creates the quaterion that rotates from "vecAxisX" to "vecAxisZ".
        vecAxisX = Vec3(1, 0, 0);
        vecAxisZ = Vec3(0, 0, 1);
        quatAtoB = VectorMath.Quat.Between(vecAxisX, vecAxisZ);

        # Creates a quaternion from the given Euler angles (in radians).
        # The angles here are 90, 60, and 30 degrees, respectively.
        quatEuler = VectorMath.Quat.EulerXYZ(1.5707, 1.0472, 0.5236);

        # Creates a quaternion from a vector representing Euler angles (in radians).
        eulerAngles = Vec3(1.5707, 1.0472, 0.5236);
        quatEuler = VectorMath.Quat.Euler(eulerAngles);

        # Gets the angle between two rotations. Quaternions must be unit length
        # for the angle to make any sense.
        angleDiff = VectorMath.Quat.Angle(quat0, quat1);

        # To combine two quaternions (combining two rotations into one), use 
        # the multiplication operator. You'll need to normalize this result to
        # guarantee a valid rotation. Order matters! quat0 * quat1 != quat1 * quat0
        combinedRot = quat0 * quat1;

        return;

Zero.RegisterComponent("QuatBasics", QuatBasics);
